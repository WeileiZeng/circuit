% 
%   cnot	   q0,A0

%   cnot	   q2,A0

%   nop	   q4
%   nop	   q4
%   nop	   q4
%   nop	   q4
%   h	   q4
%   nop	   q5
%   nop	   q5
%   nop	   q5
%   h	   q5
%   cnot	   q5,A0
%   h	   q5
%   cnot	   q4,A0
%   h	   q4
%   cnot	   q5,A0
% 
%   measure  A0
%  Time 01:
%    Gate 00 cnot(q0,A0)
%    Gate 04 nop(q4)
%    Gate 09 nop(q5)
%  Time 02:
%    Gate 01 cnot(q1,A0)
%    Gate 05 nop(q4)
%    Gate 10 nop(q5)
%  Time 03:
%    Gate 02 cnot(q2,A0)

%    Gate 03 cnot(q3,A0)
%    Gate 07 nop(q4)
%    Gate 12 h(q5)
%  Time 05:
%    Gate 08 h(q4)
%    Gate 13 cnot(q5,A0)
%  Time 06:
%    Gate 14 h(q5)
%    Gate 15 cnot(q4,A0)
%  Time 07:
%    Gate 16 h(q4)
%    Gate 17 cnot(q5,A0)

%    Gate 18 measure(A0)

% Qubit circuit matrix:
%
% q0: gAxA, n  , n  , n  , n  , n  , n  , n  , n  
% q1: n  , gBxB, n  , n  , n  , n  , n  , n  , n  
% q2: n  , n  , gCxC, n  , n  , n  , n  , n  , n  
% q3: n  , n  , n  , gDxD, n  , n  , n  , n  , n  
% q4: gAxE, gBxE, gCxE, gDxE, gExE, gFxE, gGxE, n  , n  
% q5: gAxF, gBxF, gCxF, gDxF, gExF, gFxF, gGxF, n  , n  
% A0: gAxG, gBxG, gCxG, gDxG, gExG, gFxG, gGxG, gHxG, N
